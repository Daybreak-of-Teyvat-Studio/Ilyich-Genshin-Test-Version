	MOT_Jisuan = {
		icon = generic_ignite_civil_war
		fire_only_once = no
		days_remove = 1
		days_re_enable = 0
		remove_effect = { 
			#初始粮食设置
			IF = {
				limit = {
					has_variable = MOT_liangshiVolume
				}
			}
			ELSE_IF = { set_variable = { MOT_liangshiVolume = 0 } }
			
			set_variable = { MOT_liangshiGet = num_of_civilian_factories }
			set_variable = { MOT_liangshiCost = num_of_military_factories }
			#设置一个临时变量，计算总体增减量
			set_variable = { MOT_linshibianliang = MOT_liangshiGet }
			subtract_from_variable = { MOT_linshibianliang = MOT_liangshiCost }
			set_variable = { MOT_Total_Change_ofLiangshi = MOT_linshibianliang }
			#粮食总量变化
			add_to_variable = { MOT_liangshiVolume = MOT_Total_Change_ofLiangshi }
			#根据总体增减量改变动态精神效果
			set_variable = { MOT_consum_modifier = MOT_Total_Change_ofLiangshi }
			divide_variable = { MOT_consum_modifier = -100 }
			set_variable = { MOT_cons_modifier = MOT_Total_Change_ofLiangshi }
			divide_variable = { MOT_cons_modifier = 25 }
			set_variable = { MOT_eff_modifier = MOT_Total_Change_ofLiangshi }
			divide_variable = { MOT_eff_modifier = 50 }
			#增加或更新动态民族精神
			IF = {
				limit = {
					NOT = {
						has_dynamic_modifier = MOT_Food_to_Buff_dynamic_modifier
					}
				}
				add_dynamic_modifier = { modifier = MOT_Food_to_Buff_dynamic_modifier days = -1 }
			}
			ELSE_IF = {
				force_update_dynamic_modifier = yes
			}
			
		}
	}
