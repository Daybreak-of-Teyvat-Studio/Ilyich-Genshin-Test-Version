on_actions = {
	on_capitulation = {
		effect = {
			if = {
				limit = {
					ROOT = {
						NOT = { has_country_flag = Get_Caputure }
					}		
				}
				FROM = {
					country_event = DOT_Caputure_Event.2
				}				
			}
		}
	}
	on_annex = {
		effect = {
			FROM = {
				every_unit_leader = {
					set_nationality = ROOT
				}
				every_scientist = {
					set_nationality = ROOT
				}
			}
		}
	}
}
