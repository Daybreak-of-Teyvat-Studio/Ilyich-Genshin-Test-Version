	#战争结语
	#ROOT是胜者

	on_capitulation = {
		effect = {
			#满足下面任一limit里面的条件才会刷战争结语
			#璃月
			IF = {
				limit = {
					#只要败者是璃月势力就触发
					FROM = {
						Is_LYY = yes
					}
				}
				#触发窗口
				set_global_flag = sp_warpeace_show_gui
				FROM = { save_global_event_target_as = winning_country }
				ROOT = { save_global_event_target_as = losing_country }
				add_to_variable = { global.sp_warpeace_dirty = 1 }
			}

			}

				limit = {

						FROM = {
							OR = {

						}
					}
