# 初始事件：月海幻梦
country_event = {
	id = LYY_Ganyu.1
	title = LYY_Ganyu.1.t
	desc = LYY_Ganyu.1.d
	picture = GFX_report_event_ganyu_speech
	is_triggered_only = yes
	option = {
		name = LYY_Ganyu.1.a		# 选项文本："托古改制，以仙家之治，救黎民于水火！"
		# --- 1. 确立领导人与执政党 ---
		# 确保甘雨作为角色存在
		recruit_character = LYY_GanYu
		# 设定党派支持率：月海亭派（中立主义）100%
		set_political_party = {
			ideology = neutrality
			popularity = 100
		}
		# 甘雨上位
		promote_character = LYY_GanYu
		set_politics = {
			ruling_party = neutrality
			elections_allowed = no
		}
		# 赋予甘雨专属领袖特质：观遍尘世之仙
		# 效果：适役人口+5%，陆军经验获取+0.05%，生产效率+10%，设施建造速度+5%
		add_country_leader_trait = {
			character = LYY_GanYu
			trait = LYY_trait_witness_of_mortal_world
		}
		# --- 2. 变更国家名号与外观 ---
		# 变更为：月海亭八省联合
		set_cosmetic_tag = LYY_MOONCHASE_ALLIANCE
		# --- 3. 设定国家基础数据 ---
		# 文本要求：基础稳定度80%，战争支持度50%
		set_stability = 0.8
		set_war_support = 0.5
		# 文本要求：科研槽 4
		set_research_slots = 4
		# --- 4. 添加民族精神 (Buffs/Debuffs) ---
		# 精神：但愿海波平 (稳定度+10%，军队训练速度-10%)
		add_ideas = LYY_idea_ganyu_peaceful_waves
		# 精神：秘职之弊，繁星可鉴 (工厂生产效率-30%，建造速度-20%，战争支持度-30%)
		add_ideas = LYY_idea_ganyu_secretarial_corruption
		# 精神：碎土离疆言国殇 (工厂产出-30%，资源开采效率-60%，工业研究速度-30%，适兵人口-30%)
		add_ideas = LYY_idea_ganyu_fragmented_land
		# --- 5. 设定经济与贸易制度 ---
		# 文本要求：分裂的经济体系（锁定），封闭经济
		add_ideas = LYY_split_economic_system		# 自定义经济法案，锁定
		add_ideas = closed_economy		# 贸易法案：封闭经济 (原版id)
		# --- 6. 特殊机制：战争中立期 ---
		# 注：甘雨派成立初期没有向任何势力宣战，也不能被任何势力宣战
		# 需等到国策解锁才可解锁战争，此部分可通过自定义modifier或scripted_effect实现
		add_timed_idea = {
			idea = LYY_idea_ganyu_neutrality_period
			days = 70		# 对应"向忤逆之徒，宣吾战意！"国策的30日+部分缓冲时间
		}
	}
}
