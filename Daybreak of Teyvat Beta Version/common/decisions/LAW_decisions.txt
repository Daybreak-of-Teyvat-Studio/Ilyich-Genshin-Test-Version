		icon = generic_police_action
		fire_only_once = no
		days_remove = 20
		state_target = any_controlled_state
		on_map_mode = map_and_decisions_view

			FROM = { has_dynamic_modifier = { modifier = LAW_SM_Area_Unstable2 } }
			LAW = { check_variable = { LAW_Var_Stablizing < LAW_Var_Max_Stablizing } }
		}
		custom_cost_text = decision_cost_CP_20
		custom_cost_trigger = { has_command_power > 19 }
		complete_effect = {
			hidden_effect = { add_command_power = -20 }

				limit = { LAW = { check_variable = { LAW_Var_Stablizing < LAW_Var_Max_Stablizing } } }
				custom_effect_tooltip = LAW_tt_Occupy_Stablizing
			}
			ELSE = {
				add_to_variable = { LAW_Var_Stablizing = 1 }
			}
		}
		remove_effect = {
			FROM = { remove_dynamic_modifier = { modifier = LAW_SM_Area_Unstable2 } }
			LAW  = {
				add_stability = 0.01
				custom_effect_tooltip = LAW_tt_Stabl_States

					limit = {
						has_idea = LAW_Knights_of_Ice2
