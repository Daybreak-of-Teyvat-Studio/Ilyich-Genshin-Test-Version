#将领猝死
country_event = {
	id = LYYwash.2
	title = LYYwash.2.t
	desc = LYYwash.2.d
	picture = GFX_Civil_War_of_Liyue
	
	is_triggered_only = yes
	
	option = {
		name = LYYwash.2.a
		custom_effect_tooltip = LYY_Press_down
		add_to_variable = { LYY_Work_Pressure = -30 }
		custom_effect_tooltip = LYY_One_Bad
		random_army_leader = {
			limit = {
				NOT = {
					has_unit_leader_trait = sick
				}
			}
			add_timed_unit_leader_trait = {
				trait = sick
				days = 200
			}
		}
	}
	option = {
		name = LYYwash.2.b
		custom_effect_tooltip = LYY_Press_down
		add_to_variable = { LYY_Work_Pressure = -30 }
		custom_effect_tooltip = LYY_More_Bad
		random_army_leader = {
			limit = {
				NOT = {
					has_unit_leader_trait = sick
				}
			}
			add_timed_unit_leader_trait = {
				trait = sick
				days = 30
			}
		}
		hidden_effect = {
			random_army_leader = {
				limit = {
					NOT = {
						has_unit_leader_trait = touch_fish
					}
				}
				add_timed_unit_leader_trait = {
					trait = touch_fish

				}
			}
			random_army_leader = {

					}
				}

				}
			}
		}
	}
}
