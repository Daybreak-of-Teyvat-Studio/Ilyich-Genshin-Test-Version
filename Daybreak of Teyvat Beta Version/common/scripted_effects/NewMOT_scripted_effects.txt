#优菈战后精神升级
LAW_Upgrade_National_Levels = {
	IF = {
		limit = {
			has_idea = LAW_Break_the_Ice1
		}
		swap_ideas = {
			remove_idea = LAW_Break_the_Ice1
			add_idea = LAW_Break_the_Ice2
		}
	}
	ELSE_IF = {
		limit = { has_idea = LAW_Break_the_Ice2 }
		swap_ideas = {
			remove_idea = LAW_Break_the_Ice2
			add_idea = LAW_Break_the_Ice3
		}
	}
	ELSE_IF = {
		limit = { has_idea = LAW_Break_the_Ice3 }
		swap_ideas = {
			remove_idea = LAW_Break_the_Ice3
			add_idea = LAW_Break_the_Ice4
		}
	}
	ELSE_IF = {
		limit = { has_idea = LAW_Break_the_Ice4 }
		swap_ideas = {
			remove_idea = LAW_Break_the_Ice4
			add_idea = LAW_Break_the_Ice5
		}
	}
	ELSE = {
		add_stability = 0.2
		add_war_support = 0.2
	}
}
