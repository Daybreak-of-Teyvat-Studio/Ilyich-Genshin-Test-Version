	FON_watercrisis_mission = {

		#icon = 
		activation = {
			has_completed_focus = FON_search
		}

		allowed = {
			tag = FON
		}
		
		available = {
			FON_final_date_trigger = yes
			check_variable = { FON_Percent_of_Preparation > 0.8 }
		}
		visible = {
			has_completed_focus = FON_search
		}

		days_mission_timeout = 220

		is_good = no

		timeout_effect = {
			if = {
			limit = {check_variable = { FON_percent_of_Peparation < 0.5 }}
			country_event = {days = 7 id = FON.11} #灭世事件（All gone）
			}

			else_if = {
                limit = { check_variable = { 
					FON_Percent_of_Preparation > 0.5
				    } 
				}
                limit = { check_variable = { 
					FON_Percent_of_Preparation < 0.8 
				    } 
			    }
                country_event = { days = 7 id = FON.40 }  # 内战事件
            }
		else = {
            country_event = { days = 7 id = FON.12 }  #大洪水（正常线路）
            }
        }
		complete_effect = {
			country_event = {days = 7 id = FON.12 } #大洪水（正常线路）
		}
		priority = 10000
	}

		}

        }

        days_re_enable = 30

        remove_effect = {
			add_days_mission_timeout = {
				mission = FON_watercrisis_mission

			}
			FON_increase_Neuvillette_repulse_Primordial_Sea_cost_effect = yes
		}
